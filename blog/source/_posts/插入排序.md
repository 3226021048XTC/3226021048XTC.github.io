#include <stdio.h>
int main(){
    int n;
    /*printf("输入要输入的数的数量\n");
    scanf("%d",&n); 
    printf("输入数字\n");*/
     n=4;
     int a[n+1];
      for(int i=1;i<=n;i++)scanf("%d",&a[i]);
     for(int i=1;i<=n;i++){
         int j,t,k;
         for(j=i-1;j>=1;j--){
             if(a[i]>a[j])break;
             }
             if(j!=i-1){
                 t=a[i];
                 for(k=i-1;k>j;k--)a[k+1]=a[k];
                 a[k+1]=t;
             }
     }
    for(int i=1;i<=n;i++){
         printf("%d ",a[i]); 
     }
     return 0; 
}
